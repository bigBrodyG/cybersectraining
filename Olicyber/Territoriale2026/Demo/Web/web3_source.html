<?php

if (isset($_GET['source'])) {
    die(highlight_file(__FILE__));
}

session_start();

if (!isset($_SESSION['csrf']))
    $_SESSION['csrf'] = base64_encode(openssl_random_pseudo_bytes(12));

if (isset($_POST['card_id']) && isset($_POST['csrf'])) {
    if ($_POST['csrf'] !== $_SESSION['csrf']) {
        die('Wrong CSRF!');
    }
    require_once('config.php');

    $card_id = $_POST['card_id'];
    $query1 = "SELECT user_id FROM cards WHERE card_id = '$card_id'";

    $user = query($query1);

    if (count($user) == 0) {
        die('Invalid card!');
    }

    $user = $user[0]['user_id'];

    $query = "SELECT * FROM users WHERE id = '$user'";
    $res = query($query)[0];

    if (count($res)) {

        $username_esc   = htmlspecialchars($res['username'], ENT_QUOTES, 'UTF-8');
        $signup_esc     = htmlspecialchars($res['sub'], ENT_QUOTES, 'UTF-8');
        $expiry_esc     = htmlspecialchars($res['exp'], ENT_QUOTES, 'UTF-8');

        // Frammento HTML che il JS inserirà nel div#user-info
?>
        <div class="card-result">
            <p><strong>Username:</strong> <?php echo $username_esc; ?></p>
            <p><strong>Data iscrizione:</strong> <?php echo $signup_esc; ?></p>
            <p><strong>Scadenza:</strong> <?php echo $expiry_esc; ?></p>
        </div>
    <?php
    } else {
        http_response_code(404);
    ?>
        <div class="card-result error">
            <p>Badge non trovato o non valido.</p>
        </div>
<?php
    }
    exit;
}

?>

<!DOCTYPE html>
<html lang="it">

<head>
    <meta charset="UTF-8">
    <title>QR Scanner</title>
    <script src="https://unpkg.com/html5-qrcode" type="text/javascript"></script>
    <style>
        #reader {
            width: 300px;
            margin: 20px auto;
        }

        #user-info {
            max-width: 400px;
            margin: 20px auto;
            padding: 10px;
            border: 1px solid #ccc;
        }

        .card-result.error {
            color: red;
        }
    </style>
</head>

<body>

    <h1>Verifica Card</h1>

    <div id="reader"></div>
    <div id="user-info">
        <p>Inquadra il QR code per vedere i dettagli dell'utente. O clicca <a href='?source'>qui</a> per capire come funziono.</p>
    </div>

    <input type="hidden" id="csrf_token" name="csrf_token"
        value="<?php echo htmlspecialchars($_SESSION['csrf'] ?? '', ENT_QUOTES, 'UTF-8'); ?>">

    <script>
        const userInfoDiv = document.getElementById('user-info');

        function getCsrfToken() {
            const el = document.getElementById('csrf_token');
            return el ? el.value : '';
        }

        function onScanSuccess(decodedText, decodedResult) {
            console.log("QR letto:", decodedText);

            const formData = new URLSearchParams();
            formData.append('card_id', decodedText);
            formData.append('csrf', getCsrfToken());

            fetch(window.location.href, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/x-www-form-urlencoded'
                    },
                    body: formData.toString()
                })
                .then(res => {
                    if (!res.ok) throw new Error('Errore nella richiesta');
                    return res.text();
                })
                .then(html => {
                    userInfoDiv.innerHTML = html;
                })
                .catch(err => {
                    console.error('Errore:', err);
                    userInfoDiv.innerHTML = '<p style="color:red">Errore durante la verifica della card.</p>';
                });

            html5QrcodeScanner.clear();
        }
        const html5QrcodeScanner = new Html5QrcodeScanner(
            "reader", {
                fps: 10,
                qrbox: 250
            }
        );
        html5QrcodeScanner.render(onScanSuccess);
    </script>

</body>

</html>
